    let animationFrameId;

    const loop = (time) => { 
      const deltaTime = time - lastTime; 
      if (deltaTime > 50) {  
        setAsteroids(prev => { 
          const nextAsteroids = []; 
          let livesLost = 0; 
          for (let ast of prev) { 
            if (ast.y >= 90) {  
              livesLost++; 
            } else { 
              nextAsteroids.push({ ...ast, y: ast.y + ast.speed }); 
            } 
          } 
          if (livesLost > 0) { 
            setLives(l => { 
              const newLives = l - livesLost; 
              if (newLives <= 0) setGameState('gameover'); 
              return newLives; 
            }); 
          } 
          return nextAsteroids; 
        }); 
 
        setLasers(prev => prev.map(l => ({ ...l, bottom: l.bottom + 15 })).filter(l => l.bottom < 100)); 
        lastTime = time; 
      } 
      animationFrameId = requestAnimationFrame(loop); 
    }; 
 
    animationFrameId = requestAnimationFrame(loop); 
    return () => cancelAnimationFrame(animationFrameId); 
  }, [gameState]);

    if (gameState !== 'playing') return; 
    // Diperlambat Spawner: Minimal 3 detik, maksimal 6 detik antar meteor
